\begin{equation*} \begin{split} T(21) &= T(10101) \\ & = 1 + T(1010) \\ & = 1 + (1 + T(101)) \\ & = 1 + (1 + (1 + T(10))) \\ & = 1 + (1 + (1 + (1 + T(1))))\\ & = 1 + (1 + (1 + (1 + (1 +T(0)))))\\ & = 5 \end{split} \end{equation*}
