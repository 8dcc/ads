\begin{equation*} \begin{array}{cc} f_{i+1}(e) = f_i(e) & \text{if } e \text{ is not part of the flow augmenting path} \\ f_{i+1}(e)=f_i(e)+d & \text{if } e \text{ is used in the forward direction} \\ f_{i+1}(e)=f_i(e)-d & \text{if } e \text{ is used in the reverse direction} \\ \end{array} \end{equation*}
